let scale = 1;
const universe = document.getElementById("universe");

window.addEventListener("wheel", (e) => {
  e.preventDefault();
  const delta = -e.deltaY * 0.001;
  scale += delta;
  scale = Math.min(Math.max(scale, 0.2), 2.5);
  universe.style.transform = `scale(${scale})`;
}, { passive: false });
